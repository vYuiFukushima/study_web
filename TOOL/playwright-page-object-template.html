Playwright / Page Object

Page Object パターンのコード例

main.py と pages/form.py、pages/confirm.py に分けて、テストの流れと画面操作を整理する例です。

main.py 何をするか
form.py 入力画面の操作
confirm.py 確認画面の操作
保守性を高める Page Object フォルダ構成 main.py pages/form.py pages/confirm.py

Structure

フォルダ構成例

テストの流れを書くファイルと、画面ごとの操作を書くファイルを分けます。

プロジェクト
├─ main.py
└─ pages
   ├─ form.py
   └─ confirm.py

各ファイルの役割

ファイル 役割
main.py テスト全体の流れを書く
pages/form.py 入力画面の操作を書く
pages/confirm.py 確認画面の操作や値取得を書く

main.py

テストの流れを書く

Page Objectのメソッドを呼び出し、「何をするか」が分かる形にします。

main.py

from playwright.sync_api import sync_playwright

from pages.form import FormPageObject
from pages.confirm import ConfirmPageObject


def main():

    with sync_playwright() as p:

        # ブラウザ起動
        browser = p.chromium.launch(
            headless=False
        )

        # 新しいページを作成
        page = browser.new_page()

        # 対象画面へアクセス
        page.goto("URL")

        # PageObject生成
        form_page = FormPageObject(page)
        confirm_page = ConfirmPageObject(page)

        # -----------------------------
        # 画面表示待機
        # -----------------------------
        form_page.メソッド名()

        # -----------------------------
        # テキスト入力
        # -----------------------------
        form_page.メソッド名("値")

        # -----------------------------
        # 数値入力
        # -----------------------------
        form_page.メソッド名("値")

        # -----------------------------
        # 日付入力
        # -----------------------------
        form_page.メソッド名("値")

        # -----------------------------
        # チェックボックス選択
        # -----------------------------
        form_page.メソッド名(True)

        # -----------------------------
        # セレクトボックス選択
        # -----------------------------
        form_page.メソッド名("値")

        # -----------------------------
        # ボタンクリック
        # -----------------------------
        form_page.メソッド名()

        # -----------------------------
        # 画面表示待機
        # -----------------------------
        confirm_page.メソッド名()

        # -----------------------------
        # 値取得
        # -----------------------------
        result = confirm_page.メソッド名()

        print(result)

        # -----------------------------
        # 検証
        # -----------------------------
        assert result == "期待値"

        # ブラウザ終了
        browser.close()


if __name__ == "__main__":
    main()

pages/form.py

入力画面の操作を書く

入力、選択、クリックなど、フォーム画面で使う操作をまとめます。

pages/form.py

from playwright.sync_api import Page


class FormPageObject:

    def __init__(self, page: Page):
        self.page = page

    # -----------------------------
    # 画面表示待機
    # -----------------------------
    def wait_until_networkidle(self):
        """
        画面の読み込みが終わるまで待機
        """
        self.page.wait_for_load_state("networkidle")

    # -----------------------------
    # テキスト入力
    # -----------------------------
    def input_text(self, value):
        """
        テキストを入力する
        """
        self.page.fill("セレクタ", value)

    # -----------------------------
    # 数値入力
    # -----------------------------
    def input_number(self, value):
        """
        数値を入力する
        """
        self.page.fill("セレクタ", value)

    # -----------------------------
    # 日付入力
    # -----------------------------
    def input_date(self, value):
        """
        日付を入力する
        """
        self.page.fill("セレクタ", value)

    # -----------------------------
    # Tabキー押下
    # -----------------------------
    def press_tab(self):
        """
        Tabキーを押下する
        """
        self.page.press("セレクタ", "Tab")

    # -----------------------------
    # セレクトボックス選択
    # -----------------------------
    def select_option(self, value):
        """
        プルダウンを選択する
        """
        self.page.select_option("セレクタ", value)

    # -----------------------------
    # チェックボックス選択
    # -----------------------------
    def select_checkbox(self, checked: bool):
        """
        チェックボックスをON/OFFする
        """
        checkbox = self.page.locator("セレクタ")

        if checkbox.is_checked() != checked:
            checkbox.click()

    # -----------------------------
    # ラジオボタン選択
    # -----------------------------
    def select_radio(self):
        """
        ラジオボタンを選択する
        """
        self.page.click("セレクタ")

    # -----------------------------
    # ボタンクリック
    # -----------------------------
    def click_button(self):
        """
        ボタンをクリックする
        """
        self.page.click("セレクタ")

    # -----------------------------
    # テキスト取得
    # -----------------------------
    def get_text(self):
        """
        テキストを取得する
        """
        return self.page.text_content("セレクタ")

    # -----------------------------
    # 入力値取得
    # -----------------------------
    def get_value(self):
        """
        入力値を取得する
        """
        return self.page.input_value("セレクタ")

    # -----------------------------
    # 表示確認
    # -----------------------------
    def is_visible(self):
        """
        要素が表示されているか確認する
        """
        return self.page.locator("セレクタ").is_visible()

    # -----------------------------
    # 活性確認
    # -----------------------------
    def is_enabled(self):
        """
        要素が操作可能か確認する
        """
        return self.page.locator("セレクタ").is_enabled()

    # -----------------------------
    # チェック状態確認
    # -----------------------------
    def is_checked(self):
        """
        チェック状態を確認する
        """
        return self.page.locator("セレクタ").is_checked()

pages/confirm.py

確認画面の操作を書く

表示内容の取得、状態確認、スクリーンショット取得などをまとめます。

pages/confirm.py

from playwright.sync_api import Page


class ConfirmPageObject:

    def __init__(self, page: Page):
        self.page = page

    # -----------------------------
    # 画面表示待機
    # -----------------------------
    def wait_until_networkidle(self):
        """
        画面の読み込みが終わるまで待機
        """
        self.page.wait_for_load_state("networkidle")

    # -----------------------------
    # テキスト取得
    # -----------------------------
    def get_text(self):
        """
        テキストを取得する
        """
        return self.page.text_content("セレクタ")

    # -----------------------------
    # 入力値取得
    # -----------------------------
    def get_value(self):
        """
        入力値を取得する
        """
        return self.page.input_value("セレクタ")

    # -----------------------------
    # 表示確認
    # -----------------------------
    def is_visible(self):
        """
        要素が表示されているか確認する
        """
        return self.page.locator("セレクタ").is_visible()

    # -----------------------------
    # 活性確認
    # -----------------------------
    def is_enabled(self):
        """
        要素が操作可能か確認する
        """
        return self.page.locator("セレクタ").is_enabled()

    # -----------------------------
    # チェック状態確認
    # -----------------------------
    def is_checked(self):
        """
        チェック状態を確認する
        """
        return self.page.locator("セレクタ").is_checked()

    # -----------------------------
    # 件数取得
    # -----------------------------
    def get_count(self):
        """
        要素数を取得する
        """
        return self.page.locator("セレクタ").count()

    # -----------------------------
    # 属性値取得
    # -----------------------------
    def get_attribute(self):
        """
        属性値を取得する
        """
        return self.page.get_attribute(
            "セレクタ",
            "属性名"
        )

    # -----------------------------
    # スクリーンショット取得
    # -----------------------------
    def take_screenshot(self):
        """
        スクリーンショットを取得する
        """
        self.page.screenshot(
            path="保存先ファイル名.png"
        )

    # -----------------------------
    # ボタンクリック
    # -----------------------------
    def click_button(self):
        """
        ボタンをクリックする
        """
        self.page.click("セレクタ")
